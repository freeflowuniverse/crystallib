struct DockerEngine #

struct DockerEngine {
	name string
pub mut:
	node                 builder.Node        [str: skip]
	sshkeys_allowed      []string // all keys here have access over ssh into the machine, when ssh enabled
	images               []DockerImage
	containers           []DockerContainer
	buildpath            string
	docker_registry_user string
	localonly            bool = true
	cache                bool = true
	push                 bool
	platform             []BuildPlatformType
	registries           []DockerRegistry // one or more supported DockerRegistrys
}
